total=df["总分"]
print("分数分布")
cut=pd.cut(total.values,[0,30,40,50,60,70,80,90,100],include_lowest=True)
pie=total.groupby(cut).count()
print(pie)
plt.pie(pie.values,labels=pie.index,autopct = '%3.1f%%')
plt.tight_layout()
plt.show()
plt.barh(pie.index.astype(str),pie.values)
for x,y in zip(pie.values,range(len(pie))):
    plt.text(x,y,x)
plt.tight_layout()
plt.show()
